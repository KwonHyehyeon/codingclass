➰ 자바스크립트 활용

➿ let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop; : scrollTop은 셋 중 하나, Y좌표값
나머지는 주석을 통해 확인해주세요!

∙ 사이드 메뉴 동그라미를 클릭했을 때의 효과 

document.querySelectorAll("#parallax__dot a").forEach( el => {
    el.addEventListener("click", e => {
        // a 링크 기능 차단
        // 이걸 안 해주면 부드럽게 가지 않고 뚝뚝 끊겨서 감 : a태그의 기본 동작
        // a 태그를 눌렀을때도 href 링크로 이동하지 않게 하기 위해 preventDefault를 사용하는데 이 기능을 통해 a태그의 기본 동작은 막아주고
        e.preventDefault();
        
        //getAttribute속성에 href를 가져와 이 href를 출력할 때 부드럽게 출력하게 한다는 의미(부드러운 효과를 주기 위해선 a태그의 기본 동작을 막아주어야 하는 것) 
        document.querySelector(el.getAttribute("href")).scrollIntoView({behavior: "smooth"})
    });
});

∙ 스크롤 했을 때 좌표값 구하기
∙ 스크롤 했을 때 사이드 메뉴 해당 페이지에 스크롤이 도달했을 때 active 활성화시키기


window.addEventListener("scroll", () => {
    let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop; 

    //#parallax__info의 span(숫자나오는 부분)을 텍스트로 출력하되 소수점 뒤는 없이
    document.querySelector("#parallax__info span").innerText = Math.floor(scrollTop); //Math.floor() : 버림 : 소수점 뒤에 다 버림

    document.querySelectorAll(".content__item").forEach((e, i) => {
        //scrollTop이 요소의 문서기준의 Y축값의 10을 뺀 값보다 크거나 같다면 : 스크롤이 해당페이지보다 더 내려갔거나 해당페이지라면
        if(scrollTop >= e.offsetTop-10){
            document.querySelectorAll("#parallax__dot li").forEach(li => {
                //li에 active 모두 제거
                li.classList.remove("active");
            });
            //그러나 해당 li에는 active활성화
            document.querySelector("#parallax__dot li:nth-child("+(i+1)+")").classList.add("active");
        }
    })
});
