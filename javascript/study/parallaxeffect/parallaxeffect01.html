➰ 자바스크립트 활용

1. window.addEventListener("scroll", () => {}) : (스크롤 값을 설정하기 위해서) 스크롤이 됐을 때
   ➿ let scrollTop = window.pageYOffset || window.scrollY || document.documentElement.scrollTop; : scrollTop은 셋 중 하나, Y좌표값
   ➿ scroll.innerText = Math.round(scrollTop); : 스크롤 값을 문자값으로 나오게 하되 반올림 하여 값이 나옴
   ➿ for문과 forEach문을 이용하여 scrollTop(스크롤할때 Y축값)이 Section의 높이 좌표값, 즉 div박스 상단 Y축 좌표값이랑 같거나 클 때 메뉴 넘어감 표현
2. 마우스 클릭 효과를 막고 스크롤 이동으로 부드럽게 href메뉴가 나타나도록 함.

document.querySelectorAll("#parallax__nav li a").forEach(li => {
    li.addEventListener("click", (e) => {
        //클릭하면 이벤트 막아줌(바로 딱 나오게 하는게 아니라 그 후에 부드럽게 나오게 하기 위함)
        e.preventDefault();
        //li = li a태그들 href = secton1,2,...
        //getAttribute 속성값을 가져온다.
        document.querySelector(li.getAttribute("href")).scrollIntoView({
            //부드럽게 움직이는 효과
            behavior: "smooth"
        })
    })
})
