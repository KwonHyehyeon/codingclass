function solution(left, right) {
    let count = 0;
    for(let i = left; i<=right; i++) {
        let arr = [];
        for(let j = 0; j <= i; j++) {
            if(i % j == 0) arr.push(j)
        }
        arr.length % 2 === 0 ? count += i : count -= i
    }
    return count
}

function solution(left, right) {
    var answer = 0;
    for (let i = left; i <= right; i++) {
        if (Number.isInteger(Math.sqrt(i))) {
            answer -= i;
        } else {
            answer += i;
        }
    }
    return answer;
}
