class Solution {
    public int solution(int n) {
        int answer = 0;
        String str = "";
        while(n != 0) {
            str += n%3;
            n = n / 3;
        }
        str = reverseStr(str);
        for(int i = 0; i < str.length(); i++) {
            answer += Character.getNumericValue(str.charAt(i)) * Math.pow(3, i);
        }
        return answer;
    }
    public static String reverseStr(String s) {
        return (new StringBuffer(s)).reverse().toString();
      }
}
